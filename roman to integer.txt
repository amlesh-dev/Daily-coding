Roman numerals are represented by seven different symbols: I, V, X, L, C, D and M.

Symbol       Value
I             1
V             5
X             10
L             50
C             100
D             500
M             1000
For example, two is written as II in Roman numeral, just two one's added together. Twelve is written as, XII, which is simply X + II. The number twenty seven is written as XXVII, which is XX + V + II.

#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;

int sup(char a)
{
    if(a=='I') return 1;
    if(a=='V') return 5;
    if(a=='X') return 10;
    if(a=='L') return 50;
    if(a=='C') return 100;
    if(a=='D') return 500;
    if(a=='M') return 1000;
    return 0;
}

int main() {
    int n;
    cin>>n;
    string s;
    cin>>s;
    int ans=0,p=0;
    for(int i=n;i>=0;i--)
    {
        if(sup(s[i])>=p)
        {
            ans+=sup(s[i]);
        }
        else
        {
            ans-=sup(s[i]);
        }
        p=sup(s[i]);
    }
    cout<<ans<<endl;
    /* Enter your code here. Read input from STDIN. Print output to STDOUT */   
    return 0;
}
